August 21, 2018 
var request = require('sync-request');
try {
	const response = await request('GET', photoUrl.url);
	if(response.statusCode === 200){
		fs.writeFileSync(
			'/path/to/image',
			response.body,
			'binary'
		);
} catch (error) {
	console.log(error);
}
