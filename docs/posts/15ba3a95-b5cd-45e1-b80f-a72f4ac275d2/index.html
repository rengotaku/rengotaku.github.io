ALBのアクセスログを日別に集計

July 17, 2018 
SELECT
  COUNT(*),
  date_parse(time, '%Y-%m-%dT%H:%i:%sZ') as date
FROM
  alb_logs
WHERE
  time > '2018-07-12'
AND
  request_url like '%path%'
GROUP BY
  date_parse(time, '%Y-%m-%dT%H:%i:%sZ')
ORDER BY
  count(*) desc
