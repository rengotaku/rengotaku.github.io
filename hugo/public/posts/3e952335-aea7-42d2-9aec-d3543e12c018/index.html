日付日時を数値のみ取得したい

December 18, 2018 
function dateFormat(date) {
  var Y = date.getFullYear();
  var M = ('0' + (date.getMonth() + 1)).slice(-2);
  var D = ('0' + date.getDate()).slice(-2);
  var h = ('0' + date.getHours()).slice(-2);
  var m = ('0' + date.getMinutes()).slice(-2);
  var s = ('0' + date.getSeconds()).slice(-2);

  return Y + M + D + h + m + s;
}

dateFormat(new Date())
"20181218171030"
