```ruby
# 1日前のファイルを削除
Dir.glob('path/to/dir/**/*.*').
select{ |f| File.mtime(f) < Time.now - 1.day }.
each{ |f| FileUtils.rm(f) }

# 空のフォルダを削除
Dir.glob("path/to/dir/**").each do |fo|
  next unless FileTest.directory?(fo)
  next unless Dir.entries(fo).length == 2

  puts "#{fo} empty"
  Dir.rmdir fo
end
```
