# インストール
```
$ npm install uuid --save
```

# 利用方法
```js
const uuidv5 = require('uuid/v5');

const MY_NAMESPACE = '1b671a64-40d5-491e-99b0-da01ff1f3341';
uuidv5('Hello, World!', MY_NAMESPACE); // ⇨ '630eb68f-e0fa-5ecc-887a-7c7a62614681'
```
