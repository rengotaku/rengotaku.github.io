```html
<ul class="error_list">
  <?php foreach( $form->getGlobalErrors() as $name => $error ) : ?>
    <li><?php echo $name ?> : <?php echo $error ?></li>
  <?php endforeach ?>
  <?php $errors = $form->getErrorSchema()->getErrors() ?>
  <?php if ( count($errors) > 0 ) : ?>
    <?php foreach( $errors as $name => $error ) : ?>
      <li><?php echo $name ?> : <?php echo $error ?></li>
    <?php endforeach ?>
  <?php endif ?>
</ul>
```
