```rb
class Cat
  attr_accessor :name
  def initialize(name)
    @name = name
  end
end
 
original = Cat.new("Tama")
copied = original.clone
copied.name = "Mike"

puts original.name
=> "Tama"
puts copied.name
=> "Mike"
```
