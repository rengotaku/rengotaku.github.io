# 半角文字を全角に変換する
def convert_half_to_full(param)
 # 半角カナ
  param = NKF.nkf('-Xw', param)
 # 半角英数字
 param = param.tr('0-9a-zA-Z', '０-９ａ-ｚＡ-Ｚ')
 # 半角記号
 param = param.gsub(/[!"#$%&'()*+,-.\/:;<=>?@\[\]^_`{|}~]+/,
  '!' => '！',
  '"' => '＂',
  '#' => '＃',
  '$' => '＄',
  '%' => '％',
  '&' => '＆',
  "'" => '＇',
  '(' => '（',
  ')' => '）',
  '*' => '＊',
  '+' => '＋',
  ',' => '、',
  '-' => '−',
  '.' => '．',
  '/' => '／',
  ':' => '：',
  ';' => '；',
  '<' => '＜',
  '=' => '＝',
  '>' => '＞',
  '?' => '？',
  '@' => '＠',
  '[' => '［',
  ']' => '］',
  '^' => '＾',
  '_' => '＿',
  '`' => '｀',
  '{' => '｛',
  '|' => '｜',
  '}' => '｝',
  '~' => '〜'
 )

 param
end
