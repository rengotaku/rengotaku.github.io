```php
$fruits = ['apple', 'banana'];
$bad_fruits = 'apple';
$good_fruits = array_filter($fruits, function($fruit) use ($bad_fruits) {
  return $fruit != $bad_fruits;
});

$good_fruits; // ['banana']
```
